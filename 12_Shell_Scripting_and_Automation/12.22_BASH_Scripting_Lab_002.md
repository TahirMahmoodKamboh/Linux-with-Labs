```bash
#!/bin/bash

# Check if two arguments are provided
if [ $# -ne 2 ]; then
    echo "Usage: $0 <number> <range>"
    exit 1

number=$1
range=$2

for (( i=1; i<=range; i++ ))

    echo "$number x $i = $((number * i))"

./table.sh 3 12
```

3 x 2 = 6
...
3 x 12 = 36
```

### Notes:

* `(( ))` loop allows **dynamic ranges** (better than `{1..10}`)
* `$# -ne 2` ensures **exactly two arguments** are passed

#!/bin/bash

# ---------- COLORS ----------
RED="\e[31m"
GREEN="\e[32m"
YELLOW="\e[33m"
BLUE="\e[34m"
RESET="\e[0m"

# ---------- FUNCTIONS ----------

print_table() {
    local number=$1
    local range=$2
    local i=1

    echo -e "${GREEN}Multiplication Table of $number${RESET}"
    while [ $i -le $range ]
    do
        echo "$number x $i = $((number * i))"
        i=$((i + 1))
    done
}

auto_detect_arguments() {
    if [ $# -eq 1 ]; then
        print_table "$1" 10
        return 0
    elif [ $# -eq 2 ]; then
        print_table "$1" "$2"
        return 0
    fi
    return 1
}

show_menu() {
    echo -e "${BLUE}-------------------------------${RESET}"
    echo -e "${YELLOW} MULTIPLICATION TABLE MENU ${RESET}"
    echo -e "${BLUE}-------------------------------${RESET}"
    echo "1. Table of 2"
    echo "2. Enter number (range 10)"
    echo "3. Enter number and range"
    echo "4. Exit"
    echo -e "${BLUE}-------------------------------${RESET}"
    echo -n "Enter your choice: "
}

# ---------- AUTO-DETECT ARGUMENTS ----------
if auto_detect_arguments "$@"; then
    exit 0
fi

# ---------- MENU LOOP ----------
while true
do
    show_menu
    read choice

    case $choice in
        1)
            print_table 2 10
            ;;
        2)
            echo -n "Enter number: "
            read number
            print_table "$number" 10
            ;;
        3)
            echo -n "Enter number: "
            read number
            echo -n "Enter range: "
            read range
            print_table "$number" "$range"
            ;;
        4)
            echo -e "${RED}Exiting program...${RESET}"
            break
            ;;
        *)
            echo -e "${RED}Invalid choice. Try again.${RESET}"
            ;;
    esac

    echo
done
```
